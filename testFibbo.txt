

program fibbo;
var int : n, first,second,next,c;
main(){
    first = 0;
    second = 1;
    read(n);
    write(n);
    for c = 0 to n do {
        if(c<=1) then{
            next = c;
        }else{
            next = first + second;
            first = second;
            second = next;
        }
        write(next);
    }
}
